FORMAT dicom

# =============================================================================
# Test recipe for dicom-deid-rs
#
# Derived from the production recipes.txt, trimmed down to exercise every
# supported syntax feature while remaining small enough for integration tests.
#
# Features exercised:
#   - FORMAT declaration
#   - Full-line and inline comments
#   - %filter graylist / blacklist sections
#   - All predicates: contains, notcontains, equals, notequals,
#                     missing, empty, present
#   - Logical operators: + (AND), || (OR), inline ||
#   - Pipe-separated alternatives in values
#   - All coordinate types: coordinates, ctpcoordinates,
#                           keepcoordinates, ctpkeepcoordinates
#   - Multiple coordinate regions per label
#   - All header actions: ADD, REPLACE, REMOVE, BLANK, KEEP, JITTER
#   - All value types: literal, var:<name>, func:<name>
#   - Tag formats: keyword, bare hex (GGGGEEEE), parenthesized (GGGG,EEEE)
# =============================================================================

%filter graylist

# --- Label with AND conditions and standard coordinates ----------------------
LABEL GE CT Dose Report # adapted from CTP
  contains Modality CT
  + contains Manufacturer GE
  + contains SeriesDescription Dose Report
  coordinates 0,0,512,110

# --- Label with CTP coordinates (x,y,width,height) --------------------------
LABEL GE CT Screen Captures # adapted from CTP
  contains Modality CT
  + contains Manufacturer GE MEDICAL
  + equals Rows 512
  + present BurnedInAnnotation
  ctpcoordinates 0,0,512,25

# --- Label with multiple coordinate regions ----------------------------------
LABEL Volume Rendering with keepcoordinates # adapted from CTP
  contains Modality CT
  + contains Manufacturer GE
  + equals Rows 512
  + equals Columns 512
  + contains ImageType VOLREN
  ctpcoordinates 350,0,162,30
  ctpcoordinates 390,80,122,15
  keepcoordinates 100,100,200,200

# --- Label with OR conditions and inline || ----------------------------------
LABEL Missing or Empty Manufacturer # flags-only, no coordinates
  missing Manufacturer || empty Manufacturer

# --- Label with pipe-separated alternatives in filter value ------------------
LABEL Toshiba Aplio Ultrasound
  contains Modality US
  + contains Manufacturer TOSHIBA
  + contains ManufacturerModelName A400|A500
  + equals Rows 720
  ctpcoordinates 0,0,960,60

# --- Label with notcontains and notequals ------------------------------------
LABEL Non-Siemens Non-CT
  notcontains Manufacturer SIEMENS
  + notequals Modality CT
  coordinates 0,0,256,50

# --- Label with ctpkeepcoordinates -------------------------------------------
LABEL Siemens Ultrasound
  contains Modality US
  + contains Manufacturer Siemens
  + contains ManufacturerModelName Antares
  + equals Rows 768 # inline comment
  + equals Columns 1024
  ctpcoordinates 0,0,1024,60
  ctpkeepcoordinates 400,200,200,300

# --- Label exercising || on its own line (OR between lines) ------------------
LABEL Burned In Annotation Catchall
  contains ImageType SAVE
  + contains Modality CT
  || contains BurnedInAnnotation YES
  || empty ImageType
  coordinates 0,0,512,80

%filter blacklist

LABEL Scanned Document or Secondary # adapted from CTP
  contains ImageType Secondary
  contains Modality OT

LABEL Missing or Empty ImageType
  missing ImageType || empty ImageType

LABEL Derived or Screen Save Catchall
  contains Modality CT|MR
  + contains ImageType DERIVED|SECONDARY|SCREEN|SAVE

%header

# --- Comments and metadata markers ------------------------------------------
ADD PatientIdentityRemoved YES
ADD DeidentificationMethod DICOM_DEID_RS
ADD LongitudinalTemporalInformationModified MODIFIED

# --- KEEP: preserve critical tags from broader removal -----------------------
KEEP SpecificCharacterSet YES
KEEP SOPClassUID YES
KEEP Manufacturer YES
KEEP ManufacturerModelName YES
KEEP SeriesDescription YES
KEEP Modality YES
KEEP PatientSex YES
KEEP PatientAge YES
KEEP PatientSize YES
KEEP PatientWeight YES
KEEP BodyPartExamined YES
KEEP BurnedInAnnotation YES
KEEP DeviceSerialNumber YES
KEEP SoftwareVersions YES
KEEP StudyTime YES
KEEP SeriesTime YES
KEEP AcquisitionTime YES
KEEP ContentTime YES

# --- REPLACE with func: references ------------------------------------------
REPLACE SOPInstanceUID func:hashuid
REPLACE StudyInstanceUID func:hashuid
REPLACE SeriesInstanceUID func:hashuid
REPLACE FrameOfReferenceUID func:hashuid
REPLACE AccessionNumber func:hash_accession
REPLACE UID func:hashuid

# --- REPLACE with var: references --------------------------------------------
REPLACE PatientID var:PATIENT_ID
REPLACE PatientName var:PATIENT_NAME

# --- REPLACE with literal values (using different tag formats) ---------------
REPLACE StudyID ANONYMIZED

# --- REPLACE using bare hex tag format (GGGGEEEE) ---------------------------
REPLACE 00120063 DEIDENTIFIED # DeidentificationMethodCodeSequence

# --- REPLACE using parenthesized tag format (GGGG,EEEE) ---------------------
REPLACE (0008,0050) ANON-ACC # AccessionNumber alternate format

# --- JITTER: shift dates by a configurable offset ----------------------------
JITTER StudyDate var:DATEINC
JITTER SeriesDate var:DATEINC
JITTER AcquisitionDate var:DATEINC
JITTER ContentDate var:DATEINC
JITTER InstanceCreationDate var:DATEINC

# --- BLANK: clear value but keep tag present ---------------------------------
BLANK PatientBirthDate YES
BLANK ReferringPhysicianName YES
BLANK ContentCreatorName YES
BLANK PlacerOrderNumberImagingServiceRequest YES
BLANK FillerOrderNumberImagingServiceRequest YES

# --- REMOVE: delete tags entirely --------------------------------------------
REMOVE InstitutionName YES
REMOVE InstitutionAddress YES
REMOVE StationName YES
REMOVE StudyDescription YES
REMOVE ReferringPhysicianAddress YES
REMOVE ReferringPhysicianTelephoneNumbers YES
REMOVE PerformingPhysicianName YES
REMOVE OperatorsName YES
REMOVE PatientAddress YES
REMOVE PatientBirthTime YES
REMOVE OtherPatientIDs YES
REMOVE OtherPatientNames YES
REMOVE MedicalRecordLocator YES
REMOVE Occupation YES
REMOVE AdditionalPatientHistory YES
REMOVE PatientComments YES
REMOVE RequestAttributesSequence YES
REMOVE ImageComments YES
REMOVE OverlayData YES
REMOVE OriginalAttributesSequence YES
REMOVE ModifiedAttributesSequence YES
